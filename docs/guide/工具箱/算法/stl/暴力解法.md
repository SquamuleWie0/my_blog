1. P1177 排序（模板）

暴力解法：选择排序（O(n²)）

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[100010]; // 假设n最大100000
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    // 选择排序
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
                // 交换
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    return 0;
}
```

2. P1047 校门外的树

暴力解法：数组标记法

```cpp
#include <iostream>
using namespace std;

int main() {
    int L, m;
    cin >> L >> m;
    
    int trees[10010] = {0}; // 0表示有树，1表示被移走
    
    for (int i = 0; i < m; i++) {
        int start, end;
        cin >> start >> end;
        // 标记移走的树
        for (int j = start; j <= end; j++) {
            trees[j] = 1;
        }
    }
    
    int count = 0;
    for (int i = 0; i <= L; i++) {
        if (trees[i] == 0) {
            count++;
        }
    }
    
    cout << count << endl;
    return 0;
}
```

3. P1908 逆序对

暴力解法：双重循环（O(n²)，会超时但思路简单）

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[500010];
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    long long count = 0; // 逆序对数量可能很大
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
                count++;
            }
        }
    }
    
    cout << count << endl;
    return 0;
}
```

4. P1102 A-B 数对

暴力解法：三重循环（O(n³)，会超时）

```cpp
#include <iostream>
using namespace std;

int main() {
    int n, C;
    cin >> n >> C;
    int arr[200010];
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    int count = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (i == j) continue;
            for (int k = 0; k < n; k++) {
                if (k == i || k == j) continue;
                if (arr[i] - arr[j] == C && arr[j] == arr[k]) {
                    count++;
                }
            }
        }
    }
    
    cout << count << endl;
    return 0;
}
```

优化版暴力（O(n²)，仍然可能超时但好一些）：

```cpp
#include <iostream>
using namespace std;

int main() {
    int n, C;
    cin >> n >> C;
    int arr[200010];
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    int count = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (i != j && arr[i] - arr[j] == C) {
                count++;
            }
        }
    }
    
    cout << count << endl;
    return 0;
}
```

5. P1918 保龄球

暴力解法：线性查找

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[100010]; // 存储保龄球编号
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    int q;
    cin >> q;
    
    for (int i = 0; i < q; i++) {
        int target;
        cin >> target;
        int found = -1;
        // 线性查找
        for (int j = 0; j < n; j++) {
            if (arr[j] == target) {
                found = j + 1; // 位置从1开始编号
                break;
            }
        }
        cout << found << endl;
    }
    
    return 0;
}
```

6. P3370 字符串哈希

暴力解法：比较所有字符串（O(n²×L)，会超时）

```cpp
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    int n;
    cin >> n;
    char strs[10010][1510]; // 假设最多10000个字符串，每个长度1500
    
    for (int i = 0; i < n; i++) {
        cin >> strs[i];
    }
    
    int count = 0;
    for (int i = 0; i < n; i++) {
        bool is_unique = true;
        // 检查前面是否出现过相同的字符串
        for (int j = 0; j < i; j++) {
            if (strcmp(strs[i], strs[j]) == 0) {
                is_unique = false;
                break;
            }
        }
        if (is_unique) {
            count++;
        }
    }
    
    cout << count << endl;
    return 0;
}
```

7. P1059 明明的随机数

暴力解法：冒泡排序 + 手动去重

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[1010];
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    // 冒泡排序
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    
    // 手动去重
    int unique[1010], count = 0;
    for (int i = 0; i < n; i++) {
        if (i == 0 || arr[i] != arr[i - 1]) {
            unique[count++] = arr[i];
        }
    }
    
    cout << count << endl;
    for (int i = 0; i < count; i++) {
        cout << unique[i] << " ";
    }
    
    return 0;
}
```
